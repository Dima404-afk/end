# Blender Python API пример
import bpy
import bmesh

# Создание меша
mesh = bpy.data.meshes.new("CustomMesh")
obj = bpy.data.objects.new("CustomObject", mesh)

# Добавление в сцену
scene = bpy.context.scene
scene.collection.objects.link(obj)

# Экспорт в glTF
bpy.ops.export_scene.gltf(
  filepath="export.glb",
  export_format='GLB'
)
